        // Function to fetch verse from Bible API
        async function fetchVerseFromAPI(verseReference = null) {
            try {
                let url;
                if (verseReference) {
                    url = `${BIBLE_API_BASE}${verseReference}`;
                } else {
                    // Get random verse from popular verses
                    const randomRef = POPULAR_VERSES[Math.floor(Math.random() * POPULAR_VERSES.length)];
                    url = `${BIBLE_API_BASE}${randomRef}`;
                }

                const response = await fetch(url);
                if (!response.ok) {
                    throw new Error('API request failed');
                }

                const data = await response.json();
                return {
                    text: data.text.trim().replace(/\s+/g, ' '),
                    reference: data.reference,
                    book: data.book_name,
                    chapter: data.chapter,
                    verse: data.verse
                };
            } catch (error) {
                console.log('API failed, using fallback verse:', error);
                return getFallbackVerse();
            }
        }

        // Function to get fallback verse when API is unavailable
        function getFallbackVerse() {
            const randomIndex = Math.floor(Math.random() * FALLBACK_VERSES.length);
            return FALLBACK_VERSES[randomIndex];
        }

        // Function to get a random Bible verse (now uses API)
        async function get_random_verse() {
            return await fetchVerseFromAPI();
        }

        // Function to search for specific verse
        async function searchVerse(query) {
            try {
                // Format query for API (replace spaces with +)
                const formattedQuery = query.toLowerCase().replace(/\s+/g, '+');
                return await fetchVerseFromAPI(formattedQuery);
            } catch (error) {
                console.log('Search failed:', error);
                return getFallbackVerse();
            }
        }

        // Function to get current date in various formats
        function get_current_date() {
            var today = new Date();
            var dd = String(today.getDate()).padStart(2, '0');
            var mm = String(today.getMonth() + 1).padStart(2, '0');
            var yyyy = today.getFullYear();
            
            return {
                key: yyyy + '-' + mm + '-' + dd,
                display: today.toLocaleDateString('en-US', { 
                    weekday: 'long', 
                    year: 'numeric', 
                    month: 'long', 
                    day: 'numeric' 
                })
            };
        }

        // Function to get stored data from local storage
        function get_stored_data() {
            var data = localStorage.getItem('bibleVerseData');
            return data ? JSON.parse(data) : { lastDate: null, verse: null, viewCount: 0, history: [] };
        }

        // Function to save data to local storage
        function save_data(data) {
            localStorage.setItem('bibleVerseData', JSON.stringify(data));
        }

        // Function to show new verse (button action)
        async function showNewVerse() {
            // Show loading message
            var content = document.getElementById('content');
            content.innerHTML = '<div class="loading">🙏 Loading a new verse...</div>';
            
            try {
                var verse = await get_random_verse();
                var currentDate = get_current_date();
                var storedData = get_stored_data();
                
                // Update stored data
                storedData.lastDate = currentDate.key;
                storedData.verse = verse;
                storedData.viewCount++;
                storedData.history.unshift({
                    date: currentDate.key,
                    verse: verse,
                    timestamp: new Date().toISOString()
                });
                
                // Keep only last 30 days in history
                if (storedData.history.length > 30) {
                    storedData.history = storedData.history.slice(0, 30);
                }
                
                save_data(storedData);
                displayVerse(verse, true);
            } catch (error) {
                content.innerHTML = '<div class="error">Sorry, could not load a new verse. Please try again.</div>';
                console.error('Error loading verse:', error);
            }
        }

        // Function to share verse
        function shareVerse() {
            var storedData = get_stored_data();
            if (storedData.verse) {
                var shareText = `"${storedData.verse.text}" - ${storedData.verse.reference}`;
                
                if (navigator.share) {
                    navigator.share({
                        title: 'Daily Bible Verse',
                        text: shareText,
                    });
                } else {
                    // Fallback to clipboard
                    navigator.clipboard.writeText(shareText).then(() => {
                        alert('Verse copied to clipboard!');
                    });
                }
            }
        }

        // Function to find verses by emotion
        function findVersesByEmotion(emotion) {
            var keywords = [];
            
            // Get keywords for this emotion
            if (emotionalKeywords[emotion]) {
                keywords = emotionalKeywords[emotion];
            } else {
                // If not in predefined list, use the emotion itself as keyword
                keywords = [emotion];
            }

            var matchingVerses = [];
            
            // Search for verses containing these keywords
            keywords.forEach(function(keyword) {
                var results = EMOTIONAL_VERSES.filter(verse => 
                    verse.text.toLowerCase().includes(keyword.toLowerCase())
                );
                matchingVerses = matchingVerses.concat(results);
            });

            // Remove duplicates
            var uniqueVerses = [];
            var seen = {};
            matchingVerses.forEach(function(verse) {
                if (!seen[verse.reference]) {
                    seen[verse.reference] = true;
                    uniqueVerses.push(verse);
                }
            });

            return uniqueVerses;
        }

        // Function to show history
        function showHistory() {
            var storedData = get_stored_data();
            var historyDiv = document.getElementById('history');
            
            if (historyDiv.style.display === 'block') {
                historyDiv.style.display = 'none';
                return;
            }
            
            if (storedData.history.length === 0) {
                historyDiv.innerHTML = `
                    <h3>Verse History</h3>
                    <p>No verses viewed yet. Start by sharing how you're feeling!</p>
                `;
            } else {
                var historyHtml = '<h3>Your Recent Verses</h3>';
                storedData.history.slice(0, 10).forEach(entry => {
                    var date = new Date(entry.timestamp).toLocaleDateString();
                    var time = new Date(entry.timestamp).toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'});
                    var searchedLabel = entry.searched ? ' (searched)' : '';
                    historyHtml += `
                        <div class="history-item">
                            <strong>${entry.verse.reference}${searchedLabel}</strong><br>
                            <em>${date} at ${time}</em><br>
                            "${entry.verse.text}"
                        </div>
                    `;
                });
                historyDiv.innerHTML = historyHtml;
            }
            historyDiv.style.display = 'block';
        }

        // Function to perform search
        async function performSearch() {
            var query = document.getElementById('searchInput').value.trim();
            if (!query) {
                alert('Please enter a verse reference');
                return;
            }

            var content = document.getElementById('content');
            content.innerHTML = '<div class="loading">🔍 Searching for verse...</div>';

            try {
                var verse = await searchVerse(query);
                var currentDate = get_current_date();
                var storedData = get_stored_data();
                
                // Update stored data (but don't change daily verse)
                storedData.viewCount++;
                storedData.history.unshift({
                    date: currentDate.key,
                    verse: verse,
                    timestamp: new Date().toISOString(),
                    searched: true
                });
                
                // Keep only last 30 days in history
                if (storedData.history.length > 30) {
                    storedData.history = storedData.history.slice(0, 30);
                }
                
                save_data(storedData);
                displayVerse(verse, true);
                
                // Clear search input
                document.getElementById('searchInput').value = '';
                toggleSearch();
                
            } catch (error) {
                content.innerHTML = '<div class="error">Could not find that verse. Please check the reference and try again.</div>';
                console.error('Search error:', error);
            }
        }

            if (!deferredPrompt) return;
            
            deferredPrompt.prompt();
            const { outcome } = await deferredPrompt.userChoice;
            
            if (outcome === 'accepted') {

            }
            
            deferredPrompt = null;
            document.querySelector('button[onclick="installApp()"]')?.remove();
